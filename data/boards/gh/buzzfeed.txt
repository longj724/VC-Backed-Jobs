        function showSlides(slideshow, slideshow_newpos) {
          var slideshow_elemshow = parseInt(slideshow.dataset.slideshow_elemshow);
          isNaN(slideshow_elemshow) && (slideshow_elemshow = 1);

          var slideshow_pos = Math.max(pos_min, Math.min(slideshow_newpos, pos_max));
          var new_left;
