    function filterJobs() {
        var departmentFilter = document.getElementById("department-filter").value;

        $(".job").removeClass('hidden');

        var parentsToHide = []

        if (departmentFilter !== 'All departments') {
            $(".department").each( function( index ) {
                if ($(this).text() !== departmentFilter) {
                    parentsToHide.push( $(this).parent() );
                }
            });
        }

        _.forEach(parentsToHide, function( elem ) {
            elem.addClass('hidden');
        });

        if ($(".job.hidden").length === $('.job').length) {
            $('.no-results-container').removeClass('hidden');
        } else {
            $('.no-results-container').addClass('hidden');
        }
    }
