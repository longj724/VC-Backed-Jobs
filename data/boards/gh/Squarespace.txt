  function blockScript( script ) {
    script.type = 'javascript/blocked';
    script.parentElement.removeChild( script );
    window.blockedAssets.push( script.src );

    /* ----- Makes it work on firefox ----- */
    var beforeScriptExecuteListener = function ( event ) {
      if ( script.getAttribute( 'type' ) === 'javascript/blocked' ) {
        event.preventDefault();
      }
      script.removeEventListener( 'beforescriptexecute', beforeScriptExecuteListener );
    }

    script.addEventListener( 'beforescriptexecute', beforeScriptExecuteListener );
  }

  function blockUniversalAssets() {
